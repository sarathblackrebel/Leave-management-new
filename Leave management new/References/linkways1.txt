    function nowMoment(date, id) {
        var NowDate = new Date(date).toISOString();
        $('#moment' + id).html(moment(NowDate).from(new Date().toISOString()));
    }

    function unreadNotificationCount() {
        var is_valid_count_in_local_storage = false;
        var now = moment(new Date().toISOString()); //todays date
        var count_from_local_storage = localStorage.getItem('notification_count');
        if (count_from_local_storage) {
            var local_storage_last_updated = moment(localStorage.getItem('notification_count_updated'));
            var duration = moment.duration(now.diff(local_storage_last_updated));
            var hours = Math.round(duration.asHours());
            if (hours <= '1') {
                if (count_from_local_storage >= 1) {
                    $('.notification-globe').append("<i class=\"badge badge-notify\">" + count_from_local_storage + "</i>");
                    is_valid_count_in_local_storage = true;
                }
            }
        }
        if (is_valid_count_in_local_storage == false) {

            $.get('../libcommon/notification/unread_notification_count.php', function (response) {
                response = $.trim(response);
                if (response != '0') {
                    $('.notification-globe').append("<i class=\"badge badge-notify\">" + response + "</i>");
                }
                localStorage.setItem('notification_count_updated', new Date().toISOString());
                localStorage.setItem('notification_count', response);
            });
        }
    }
